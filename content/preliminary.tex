\paragraph{Global} 

\begin{itemize}
  \item $\mathbb{H}$ - Global hash rate
  \item $P$ - Set of pools
\end{itemize}

\paragraph{Blockchain}

\begin{itemize}
  \item $\mathbb{R}$ - Block reward
  \item $\mathbb{T}$ - Block time
  \item $\mathbb{C}$ - Market cap
\end{itemize}

\paragraph{Pool}

\begin{itemize}
  \item $C$ - total chunks
  \item $c$ - Available chunks
  \item $H$ - Hash rate
  \item $i$ - Trade intervals 
  \item $t$ - Tax
  \item $P$ - Chunk sell price
  \item $R$ - Reward per chunk
\end{itemize}

\begin{equation}
  R = \frac{\mathbb{B}}{c} \times \frac{H}{\mathbb{H}}
\end{equation}

\paragraph{Seller}

\begin{itemize}
  \item $x$ - Owned chunks
  \item $p_s$ - Sell price
\end{itemize}
